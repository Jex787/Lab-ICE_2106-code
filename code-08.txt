#include <iostream>
#include <cmath>
using namespace std;

double func(double x)
{
    return x * x * x - x * x + 2;
}

double derivFunc(double x)
{
    return 3 * x * x - 2 * x;
}

int main()
{
    double errlimit;
    cout << "Enter err limit: ";
    cin >> errlimit;

    double x;
    cout << "Enter initial value: ";
    cin >> x;

    while (true)
    {
        double h = func(x) / derivFunc(x);
        x = x - h;

        if (abs(h) < errlimit)
        {

            break;
        }
    }

    cout << "The value of the root is : " << x;
    return 0;
}



// output:
//Enter err limit: 0.0001
//Enter initial value: 2
//The value of the root is : -1

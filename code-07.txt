#include <iostream>
#include <cmath>
using namespace std;

double equation(double x)
{
    return x * x - x - 2;
}

int main()
{
    double a = 1.0, b = 3.0, errLimit, x0;

    cout << "Enter the error limit: ";
    cin >> errLimit;

    while (abs(b - a) > errLimit)
    {
        x0 = a - (equation(a) * (b - a)) / (equation(b) - equation(a));

        if (equation(x0) == 0.0)
        {
            break;
        }
        else if (equation(x0) * equation(a) < 0)
        {
            b = x0;
        }
        else
        {
            a = x0;
        }
    }

    cout << "Approximate root: " << x0 << endl;

    return 0;
}


//output:

//Approximate root: 2
